  import sys
  print(sys.getdefaultencoding())
  import os
  def get_filePaths(directory):
      """
  
      """
      file_paths=[]
      for root, directories, files in os.walk(directory):
          for filename in files:
              filepath=os.path.join(root,filename)
              file_paths.append(filepath)
  
      return file_paths
